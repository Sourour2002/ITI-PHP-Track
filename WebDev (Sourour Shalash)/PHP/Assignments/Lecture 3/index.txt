    function isArmstrong($number) {
        $sum = 0;
        $temp = $number;
        while ($temp != 0) {
            $digit = $temp % 10;
            $sum += $digit ** 3;
            $temp = (int)($temp / 10);
        }
        return $sum === $number;
    }

    function isPrime($num) {
        if ($num < 2) return false;
        for ($i = 2; $i <= sqrt($num); $i++) {
            if ($num % $i == 0) return false;
        }
        return true;
    }

    for ($i = 2; $i < 100; $i++) {
        if (isPrime($i)) {
            $sum += $i;
